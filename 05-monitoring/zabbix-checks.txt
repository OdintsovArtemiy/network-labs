# ============================================
# Zabbix проверки для сетевого оборудования
# Используются команды zabbix_get
# ============================================

# ============================================
# 1. Базовые проверки доступности
# ============================================

# Проверка, отвечает ли коммутатор на SNMP
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o system.sysName.0

# Uptime устройства (как давно работает)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o system.sysUpTime.0

# ============================================
# 2. Мониторинг производительности
# ============================================

# Загрузка CPU (OID для Cisco)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.4.1.9.9.109.1.1.1.1.5.1

# Свободная память
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.4.1.9.9.48.1.1.1.5.1

# Температура (если поддерживается)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.4.1.9.9.91.1.1.1.1.4.1

# ============================================
# 3. Мониторинг интерфейсов
# ============================================

# Статус интерфейса (1=up, 2=down)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.2.1.2.2.1.8.1

# Входящий трафик (байт/сек)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.2.1.2.2.1.10.1

# Исходящий трафик
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.2.1.2.2.1.16.1

# Ошибки на интерфейсе (признак проблем с кабелем)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.2.1.2.2.1.14.1

# ============================================
# 4. Мониторинг VLAN и MAC (для InfraSec)
# ============================================

# Количество VLAN на устройстве
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.4.1.9.9.68.1.2.2.1.2 | wc -l

# Количество MAC-адресов (резкий рост = возможная атака)
zabbix_get -s 192.168.1.1 -p 161 -O snmpv2 -c rostring -o 1.3.6.1.2.1.17.4.3.1.1 | wc -l

# ============================================
# 5. Полезные OID для запоминания
# ============================================
# 
# .1.3.6.1.2.1.1.1.0          - описание устройства
# .1.3.6.1.2.1.1.3.0          - uptime
# .1.3.6.1.2.1.1.5.0          - hostname
# .1.3.6.1.2.1.2.2.1.2        - имя интерфейса
# .1.3.6.1.2.1.2.2.1.8        - статус интерфейса
# .1.3.6.1.2.1.2.2.1.10       - входящий трафик
# .1.3.6.1.2.1.2.2.1.16       - исходящий трафик
# .1.3.6.1.4.1.9.9.109.1.1.1.1.5.1 - CPU Cisco
